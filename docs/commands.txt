# Quick Reference Commands

# If you did not install the package, prefix with PYTHONPATH=src

# 1. Download channel videos metadata (auto-saves to data/<channel>/videos.json)
PYTHONPATH=src python -m ytce videos @channelname
PYTHONPATH=src python -m ytce videos @channelname --max-videos 10

# 2. Download comments for single video (auto-saves to data/<video_id>/comments.jsonl)
PYTHONPATH=src python -m ytce comments VIDEO_ID
PYTHONPATH=src python -m ytce comments VIDEO_ID --limit 100

# 3. Download channel + all comments (auto-saves to data/<channel>/)
PYTHONPATH=src python -m ytce channel-comments @channelname
PYTHONPATH=src python -m ytce channel-comments @channelname --max-videos 5 --per-video-limit 50

# Custom output paths (optional)
PYTHONPATH=src python -m ytce videos @channelname -o custom/path.json
PYTHONPATH=src python -m ytce comments VIDEO_ID -o custom/path.jsonl
PYTHONPATH=src python -m ytce channel-comments @channelname --out-dir custom/dir

# All data exports go to data/ folder by default:
# data/<channel>/videos.json
# data/<channel>/comments/0001_<videoId>.jsonl
# data/<video_id>/comments.jsonl
